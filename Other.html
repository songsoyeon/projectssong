css

display: flex;지정된 Flex Container는 Block 요소와 같은 성향(수직 쌓임)
flex-direction: column;Items를 수직축(위에서 아래로)으로 표시
justify-content: none;주 축(main-axis)의 정렬 방법을 설정
align-items: none;교차 축(cross-axis)에서 Items의 정렬 방법을 설정

function showmain() {
        document.getElementById("fig").src = "img/main.jpg";
        document.getElementById("desc").innerHTML = "이름: 송소연<br>학번: 91810521";
}
fig 아이디의 src 속성에 img/main.jpg를 넣어주는 함수
